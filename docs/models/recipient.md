# Recipient

Recipient data for a shipment, including full postal address.

## Example Usage

```typescript
import { Recipient } from "@postivo/postivo-client/models";

let value: Recipient = {
  name: "Jan Nowak",
  name2: "Firma Testowa Sp. z o.o.",
  address: "ul. Aleje Jerozolimskie",
  homeNumber: "31",
  flatNumber: "2",
  postCode: "00-999",
  city: "Warszawa",
  phoneNumber: "+48999999999",
  postscript: "Do rąk własnych",
};
```
