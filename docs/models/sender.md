# Sender

Sender data for the shipment.

## Example Usage

```typescript
import { Sender } from "@postivo/postivo-client/models";

let value: Sender = {
  name: "Jan Nowak",
  name2: "Firma Testowa Sp. z o.o.",
  address: "ul. Aleje Jerozolimskie",
  homeNumber: "31",
  flatNumber: "2",
  postCode: "00-999",
  city: "Warszawa",
  country: "PL",
};
```
